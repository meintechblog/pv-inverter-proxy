         - if self._session is None or self._profile is None: return PollResult([], [], success=False, error="Not connected")

         - return PollResult(common_regs, inverter_regs, success=True)
       - def _track_energy(self, raw_energy_wh: float) -> float:
         - if raw_energy_wh < self._last_energy_raw_wh: self._energy_offset_wh += self._last_energy_raw_wh
         - self._last_energy_raw_wh = raw_energy_wh
         - return raw_energy_wh + self._energy_offset_wh
       - def _encode_model_103(self, data: ShellyPollData, energy_wh: float) -> list[int]:
         - regs = [0] * 52
         - regs[0] = 103 (DID), regs[1] = 50 (Length)
         - regs[2] = int(round(data.current_a * 100)) -- AC current total, SF=-2
         - regs[3] = int(round(data.current_a * 100)) -- Phase A (single-phase)
         - regs[4] = 0, regs[5] = 0 -- Phase B/C
         - regs[6] = _int16_as_uint16(-2) -- AC Current SF
         - regs[10] = int(round(data.voltage_v * 10)) -- AC Voltage AN, SF=-1
         - regs[13] = _int16_as_uint16(-1) -- AC Voltage SF
         - regs[14] = int(round(data.power_w)) & 0xFFFF -- AC Power, SF=0
         - regs[15] = 0 -- AC Power SF
         - regs[16] = int(round(data.frequency_hz * 100)) -- Frequency, SF=-2
         - regs[17] = _int16_as_uint16(-2) -- Frequency SF
         - energy_int = int(round(energy_wh))
         - regs[24] = (energy_int >> 16) & 0xFFFF -- Energy high word
         - regs[25] = energy_int & 0xFFFF -- Energy low word
         - regs[26] = 0 -- Energy SF
         - regs[27] = 0, regs[28] = _int16_as_uint16(-2) -- DC Current = 0
         - regs[29] = 0, regs[30] = _int16_as_uint16(-1) -- DC Voltage = 0
         - regs[31] = 0, regs[32] = 0 -- DC Power = 0
         - regs[33] = int(round(data.temperature_c * 10)) -- Temp, SF=-1
         - regs[37] = _int16_as_uint16(-1) -- Temp SF
         - regs[38] = 4 if data.relay_on else 2 -- Status: MPPT or SLEEPING
         - return regs

         - regs[0] = COMMON_DID, regs[1] = COMMON_LENGTH
         - regs[2:18] = encode_string("Shelly", 16)  -- Manufacturer
         - regs[18:34] = encode_string(self._name, 16) -- Model
         - mac = self._device_info.get("mac", "")
         - regs[50:66] = encode_string(mac, 16) -- Serial
         - regs[66] = PROXY_UNIT_ID
         - return regs
       - def get_static_common_overrides(self) -> dict[int, int]:

       - def get_model_120_registers(self) -> list[int]:

         - regs[0] = NAMEPLATE_DID (120), regs[1] = NAMEPLATE_LENGTH (26)
         - regs[2] = 4 (DERTyp = PV), regs[3] = self._rated_power (WRtg), regs[4] = 0 (WRtg_SF)
         - return regs
       - async def write_power_limit(self, enable: bool, limit_pct: float, *, force: bool = False) -> WriteResult:
         - return WriteResult(success=True) -- no-op, Shelly cannot do percentage limiting

         - if self._session is not None and not self._session.closed: await self._session.close()
         - self._session = None
         - log.info("shelly_plugin_closed", host=self._host)
